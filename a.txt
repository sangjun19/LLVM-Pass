std::map<const BasicBlock*, unsigned> sucCnts;
for (BasicBlock &BB : F) {
    const Instruction *ter = BB.getTerminator();
    if(ter) numSucs = ter->getNumSuccessors();
    else numSucs = 0;
    sucCnts[&BB] = numSucs;
}

const BasicBlock *sucCan = nullptr;
unsigned maxSuc = 0;
for (auto const& [block, count] : sucCnts) {
    if (count > maxSuc) {
        maxSuc = count;
        sucCan = block;
    }
}
